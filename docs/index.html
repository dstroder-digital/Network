// 5) (Nice-to-have) If the PMTiles file contains bounds/center, fly to it. // pmtiles.js lets us read headers/metadata. (async () => { try { const p = new pmtiles.PMTiles(EDGES_PMTILES.replace("pmtiles://","")); const header = await p.getHeader(); // may include center/zoom/bounds // Prefer center from header if present if (header && Array.isArray(header.center) && header.center.length >= 2) { const [lng, lat, z] = header.center; map.once("load", () => map.easeTo({ center: [lng, lat], zoom: z ?? 10 })); } else if (header && Array.isArray(header.bounds) && header.bounds.length === 4) { const b = header.bounds; // [w,s,e,n] map.once("load", () => map.fitBounds([[b[0], b[1]], [b[2], b[3]]], { padding: 40 })); } } catch (e) { // Non-fatal; we still show the map at the fallback center console.warn("Could not read PMTiles header/metadata:", e); } })();
